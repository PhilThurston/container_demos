## insert timeout
https://linkerd.io/2/tasks/configuring-timeouts/
kubectl edit sp/authors.default.svc.cluster.local

insert timeout into spec
  - condition:
      method: HEAD
      pathRegex: /authors/[^/]*\.json
    isRetryable: true
    timeout: 10ms
    name: HEAD /authors/{id}.json
